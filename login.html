Script d'initialisation SQL

Copiez ce script et exécutez-le dans le "SQL Editor" de votre dashboard Supabase.

-- 1. Create Profiles table
CREATE TABLE IF NOT EXISTS public.profiles (
  id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
  updated_at TIMESTAMP WITH TIME ZONE,
  tier TEXT DEFAULT 'free',
  dashboard_config JSONB DEFAULT '[]'::jsonb,
  ha_url TEXT,
  ha_token_enc TEXT,
  ha_entity_energy TEXT,
  theme TEXT DEFAULT 'default'
);

-- 2. Create Rooms table
CREATE TABLE IF NOT EXISTS public.rooms (
  id TEXT NOT NULL,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE NOT NULL,
  name TEXT NOT NULL,
  PRIMARY KEY (id, user_id)
);

-- 3. Create Entities table
CREATE TABLE IF NOT EXISTS public.entities (
  haid TEXT NOT NULL,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE NOT NULL,
  name TEXT,
  type TEXT,
  variant TEXT,
  roomid TEXT,
  PRIMARY KEY (haid, user_id)
);

-- Enable RLS
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.rooms ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.entities ENABLE ROW LEVEL SECURITY;

-- Set Policies
CREATE POLICY "Users can see own profile" ON public.profiles FOR SELECT USING (auth.uid() = id);
CREATE POLICY "Users can update own profile" ON public.profiles FOR UPSERT WITH CHECK (auth.uid() = id);
CREATE POLICY "Users can manage own rooms" ON public.rooms FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Users can manage own entities" ON public.entities FOR ALL USING (auth.uid() = user_id);
